
eg.2实现一个函数找出一维数组的最大值和最小值  
```void find_min_max(int a[], int n, int* min, int* max)  
{  
	*min = *max = a[0];  
	for(int i = 1; i < n; i++) {  
		if(*max < a[i]) {  
			*max = a[i];  
		}  
		if(*min > a[i]) {  
			*min = a[i];  
		}  
	}  
}  
int main()  
{  
	int a[5] = {4, 5, 1, 3, 2};  
	int min, max;  
	find_min_max(a, 5, &min, &max);  
}```
