gcd竟然有内部调用函数？！
```c++
#include<bits/stdc++.h>

using namespace std;

int n, a, b;

int main(void){
    cin >> n;
    while(n--){
        cin >> a >> b;
        cout << __gcd(a, b) << endl;
    }
    return 0;
}
```
