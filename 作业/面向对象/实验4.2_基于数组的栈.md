package test1;
import java.util.Scanner;

public class MyStack {
	private int idx = 0;
	private String[] data;
	private int max_size;
	
	public void setSize(int size) { // 设置stack大小，并进行初始化
		max_size = size;
		data = new String[max_size];
	}
	
	public String pop() { // pop 函数1
		if(idx > 0) {
			return data[idx--];
		}else {
			System.out.println("Sorry, the stack is empty.");
		return "NULL";
		}
	}
	
	public boolean push(String s) { // push 函数
		if(idx < max_size) {
			idx++;
			data[idx] = s;
			return true;
		}else
			return false;
	}
	
	public String getTop() { // getTop 函数
		if(idx > 0)
			return data[idx];
		else
			System.out.println("Sorry, the stack is empty.");
		return "NULL";
	}
	
	public int getLength() { // getLength 函数
		return idx;
	}
	
	public boolean isEmpty(){ // 判断 stack 是否为空
		if(idx == 0) {
			return true;
		}else
			return false;
	}
	
	public boolean Clear() { // 清空stack
	
		for(int i = 0; i < max_size; i++)
			data[i] = "NULL"; // 清空stack
			
		System.out.println("The stack has been cleared.");
		idx = 0;
		return true;
	}
	
	public static void main(String[] args) {
		MyStack mystack = new MyStack(); // 创建实例对象
		Scanner input = new Scanner(System.in);
		
		System.out.println("Please set the max size: ");
		mystack.setSize(input.nextInt()); // 设置栈的最大存储尺寸
		
		if(mystack.isEmpty()) {
			System.out.println("The stack is empty.");
		}else{
			System.out.print("The stack is not empty");
		} // 测试empty函数
		
		mystack.push("Lily");
		System.out.println("The top element of the stack is " + mystack.getTop()); // 测试push与getTop函数
		System.out.println("The length of the stack is " + mystack.getLength());

		// 测试pop函数
		mystack.push("Jack"); 
		System.out.println("The stack pops the element: " + mystack.pop());
		
		mystack.pop();
		mystack.pop(); // 测试无内容后pop的反馈

		mystack.push("Tom");
		mystack.Clear(); // 测试清空数组
		if(mystack.isEmpty()) {
			System.out.println("The stack is empty.");
		}else{
			System.out.print("The stack is not empty");
		} // 测试empty判断
	}
}

# 算法原理概述
## 栈
使用数组来模拟栈，主要操作即为使用一个 idx 作为标记，来表示栈的当前尺寸，栈顶元素等，data\[idx] 对应的元素即为栈顶元素。添加元素则令idx ++，弹出元素则令idx-- ，即可使 idx 始终能对栈进行表示。
清空栈只需要对栈内元素遍历，使其全部存放 "NULL" ，并将 idx 归零即可。
